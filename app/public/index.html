Samples

Single queue center

Used to model single processor systems, individual devices or software process in a computer system. Notation: M/M/1, M/D/1, M/G/1


        use pdq;

        $arrivRate = 0.75;
        $servTime  = 1.0;

        pdq::Init("Open Network with M/M/1");
        pdq::CreateOpen("Work", $arrivRate);

        pdq::CreateNode("Server", $pdq::CEN, $pdq::FCFS);  
        pdq::SetDemand("Server", "Work", $servTime);

        pdq::Solve($pdq::CANON);
        pdq::Report();

Parallel queue centers

Used to describe a set of parallel centers or devices in a system where each center has its own input queue. Individually each service center is M/M/1, M/D/1, M/G/1.


        use pdq;

        $servers = 5;

        $arrivRate = 0.75;
        $servTime  = 1.0;

        pdq::Init("Open Network with n(M/M/1)");
        pdq::CreateOpen("Work", $arrivRate);

        for($d=1; $d<=$servers; $d++){
            $sname = "Server-" . $d;
            pdq::CreateNode($sname, $pdq::CEN, $pdq::FCFS);  
            pdq::SetDemand($sname, "Work", $servTime / $servers);
        }

        pdq::Solve($pdq::CANON);
        pdq::Report();

Multiserver single queue

Used to model multiprocessor systems or devices that have several identical servers and all jobs waiting for these servers are kept in one queue. Notation: M/M/m, M/D/m, M/G/m.


        use pdq;
        
        $servers = 5;

        $arrivRate = 0.75;
        $servTime  = 1.0;

        pdq::Init("Open Network with M/M/m)");
        pdq::CreateOpen("Work", $arrivRate);

        pdq::CreateMultiNode($servers, "5servers", $pdq::CEN, $pdq::FCFS);  
        pdq::SetDemand("5servers", "Work", $servTime);

        pdq::Solve($pdq::CANON);
        pdq::Report();
